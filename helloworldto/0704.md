# 1089. 复写零 #
思路：新建数组 保存元素

	class Solution {
	    public void duplicateZeros(int[] arr) {
	        int len = arr.length;
	        int index = 0;
	               int data[] = new int[len];
	        for(int i = 0;i < arr.length && index < len;i++){
	            
	            if(arr[i] == 0) 
	                data[index++] = 0;
	                
	            if(index < len)
	             data[index++] = arr[i];
	        }
	        for(int i = 0; i < len;i++){
	            arr[i] = data[i];
	        }
	        
	    }
	
	    
	}


# 面试题 08.11. 硬币 #
思路：背包问题


	class Solution {
	    public int waysToChange(int n) {
	        int[] dp = new int[1+n];
	        int[] coins = {1,5,10,25};
	        dp[0] = 1;
	        for(int coin : coins){
	            for(int i = coin; i <=n;i++)
	                dp[i] =  (dp[i] + dp[i-coin])%1000000007;;
	
	        }
	        return dp[n];
	    }
	}
